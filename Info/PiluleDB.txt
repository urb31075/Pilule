
CREATE TABLE Pilule.GoodsDictionary (
  Id INT(11) NOT NULL AUTO_INCREMENT,
  Name VARCHAR(50) NOT NULL,
  Price DECIMAL(8, 2) NOT NULL,
  Comment VARCHAR(255) DEFAULT NULL,
  PRIMARY KEY (Id)
)
ENGINE = INNODB
AUTO_INCREMENT = 6
AVG_ROW_LENGTH = 3276
CHARACTER SET koi8r
COLLATE koi8r_general_ci
ROW_FORMAT = DYNAMIC;

CREATE TABLE Pilule.StockBalance (
  Id INT(11) NOT NULL AUTO_INCREMENT,
  GoodsId INT(11) NOT NULL,
  Amount DECIMAL(8, 3) NOT NULL,
  PRIMARY KEY (Id),
  CONSTRAINT FK_StockBalance_GoodsId FOREIGN KEY (GoodsId)
    REFERENCES Pilule.GoodsDictionary(Id) ON DELETE NO ACTION ON UPDATE RESTRICT
)
ENGINE = INNODB
AUTO_INCREMENT = 8
AVG_ROW_LENGTH = 2340
CHARACTER SET koi8r
COLLATE koi8r_general_ci
ROW_FORMAT = DYNAMIC;

CREATE TABLE Pilule.Version (
  info VARCHAR(25) DEFAULT NULL
)
ENGINE = INNODB
CHARACTER SET koi8r
COLLATE koi8r_general_ci
ROW_FORMAT = DYNAMIC;
