CREATE TABLE Dupel.StockBalance (
  Id INT(11) NOT NULL AUTO_INCREMENT,
  GoodsId INT(11) NOT NULL,
  Amount DECIMAL(8, 3) NOT NULL,
  PRIMARY KEY (Id),
  CONSTRAINT FK_StockBalance_GoodsId FOREIGN KEY (GoodsId)
    REFERENCES Pilule.GoodsDictionary(Id) ON DELETE NO ACTION ON UPDATE RESTRICT
)
ENGINE = INNODB
AUTO_INCREMENT = 1
AVG_ROW_LENGTH = 2340
CHARACTER SET koi8r
COLLATE koi8r_general_ci
ROW_FORMAT = DYNAMIC;
